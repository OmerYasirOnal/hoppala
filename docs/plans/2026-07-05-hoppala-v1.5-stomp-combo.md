# Hoppala v1.5 — Stomp Combo + Score Pop Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Reward chained stomps — consecutive kills within a window build a combo whose multiplier grows a score bonus (added to the leaderboard score), and each kill spawns a floating "+N ×C" pop.

**Architecture:** The combo + score bonus live in the pure seeded sim (deterministic, daily-fair): `world.combo`/`stompBonus`/`comboEndsAt` updated on each stomp, plus a transient `world.stompFx` list of this step's kills. `main.ts` adds `stompBonus` to the score and feeds `stompFx` to the renderer per update; the renderer owns a cosmetic floating-pop list. Reuses the v1.4 `'stomp'` sites.

**Tech Stack:** Vanilla TypeScript, Canvas 2D, Vitest 4.

**Spec:** `docs/specs/2026-07-05-hoppala-v1.5-stomp-combo-design.md`

## Global Constraints

- **Deterministic core:** combo + bonus are pure seeded sim state — a fixed daily seed replays byte-identically (kills included). The score pop is a cosmetic renderer effect (its timing never affects score/state).
- **Reachability + determinism tests stay green.** World is y-down. Core entry < 60KB gzip. 60fps. No new deps, no assets, no i18n.
- **Score = `floor(maxAltitude/10) + stompBonus`.** Bonus is modest (altitude dominates) and stays under the online 100000 cap.
- **Per-step consumption:** `world.stompFx` is fed to the renderer in the composition UPDATE (like `world.events`), never read once in render — so pops aren't missed on frame catch-up.
- **Exact values:** `TUNING.stompBonus = 3` (bonus meters per combo level), `TUNING.comboWindow = 1.5` (s); pop `POP_DURATION = 0.8` (s).
- tsconfig `strict` + `noUncheckedIndexedAccess`. Pristine tests. Conventional commits. pnpm only. Browser automation uses Brave.

describe('v1.5 stomp combo', () => {
  const enemy = (over: Partial<import('../src/game/types').Enemy> = {}) => ({
    id: 1, x: 200, y: 0, baseX: 200, amp: 0, speed: 0, phase: 0, dead: false, ...over,

  const stomp = (w: ReturnType<typeof world>, ey: number): boolean => {
    w.enemies = [enemy({ x: 200, y: ey })];
    w.player.x = 200;
    w.player.y = ey - TUNING.enemyR - TUNING.playerR - 1;
    w.player.vy = 300;
    let done = false;
    for (let i = 0; i < 6 && !done; i++) {
      step(w, 200, mulberry32(70 + i));
      done = w.events.includes('stomp');
    }
    return done;
  };

  it('a stomp starts the combo at 1 and adds stompBonus to the score bonus', () => {
    const w = world();
    w.platforms = [];
    expect(stomp(w, 0)).toBe(true);
    expect(w.combo).toBe(1);
    expect(w.stompBonus).toBe(TUNING.stompBonus);
    expect(w.stompFx.length).toBe(1);
    expect(w.stompFx[0]!.combo).toBe(1);
    expect(w.stompFx[0]!.bonus).toBe(TUNING.stompBonus);
  });

  it('chains the combo and escalates the bonus on quick successive stomps', () => {
    const w = world();
    w.platforms = [];
    expect(stomp(w, 0)).toBe(true);
    expect(w.combo).toBe(1);
    expect(stomp(w, -400)).toBe(true); // a few steps later — well within comboWindow
    expect(w.combo).toBe(2);
    expect(w.stompBonus).toBe(TUNING.stompBonus * 1 + TUNING.stompBonus * 2); // 3 + 6 = 9

    w.platforms = [];
    expect(stomp(w, 0)).toBe(true);
    w.comboEndsAt = w.time - 0.001; // force the window to have already lapsed
    expect(stomp(w, -400)).toBe(true);
    expect(w.combo).toBe(1);

/** Update the stomp combo + score bonus and record the kill for the score pop. */
function registerStomp(world: World, e: Enemy): void {
  world.combo = world.time <= world.comboEndsAt ? world.combo + 1 : 1;
  world.comboEndsAt = world.time + TUNING.comboWindow;
  const bonus = TUNING.stompBonus * world.combo;
  world.stompBonus += bonus;
  world.stompFx.push({ x: e.x, y: e.y, bonus, combo: world.combo });
}
